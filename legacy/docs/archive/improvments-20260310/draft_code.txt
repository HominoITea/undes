const SYMBOL_PATTERNS = {
  js: [
    { type: 'function', re: /\bfunction\s+([A-Za-z_$][\w$]*)\s*\(/g },
    { type: 'class', re: /\bclass\s+([A-Za-z_$][\w$]*)\b/g },
    { type: 'const-fn', re: /\bconst\s+([A-Za-z_$][\w$]*)\s*=\s*\([^)]*\)\s*=>/g },
  ],
  ts: [
    { type: 'function', re: /\bfunction\s+([A-Za-z_$][\w$]*)\s*\(/g },
    { type: 'class', re: /\bclass\s+([A-Za-z_$][\w$]*)\b/g },
    { type: 'interface', re: /\binterface\s+([A-Za-z_$][\w$]*)\b/g },
    { type: 'type', re: /\btype\s+([A-Za-z_$][\w$]*)\s*=/g },
    { type: 'method', re: /\b(?:public|private|protected)?\s*([A-Za-z_$][\w$]*)\s*\([^)]*\)\s*\{/g },
  ],
  py: [
    { type: 'class', re: /^\s*class\s+([A-Za-z_][\w]*)\s*(?:\(|:)/gm },
    { type: 'function', re: /^\s*def\s+([A-Za-z_][\w]*)\s*\(/gm },
  ],
  java: [
    { type: 'class', re: /\bclass\s+([A-Za-z_][\w]*)\b/g },
    { type: 'method', re: /\b(?:public|private|protected)?\s*(?:static\s+)?[\w<>\[\]]+\s+([A-Za-z_][\w]*)\s*\([^)]*\)\s*\{/g },
  ],
  cs: [
    { type: 'class', re: /\bclass\s+([A-Za-z_][\w]*)\b/g },
    { type: 'method', re: /\b(?:public|private|protected|internal)?\s*(?:static\s+)?[\w<>\[\],?]+\s+([A-Za-z_][\w]*)\s*\([^)]*\)\s*\{/g },
  ],
};

function resolveIndexConfig(contextConfig = {}) {
  const cfg = contextConfig.codeIndex || {};
  return {
    enabled: cfg.enabled !== false,
    outputPath: cfg.outputPath || '.code_index.json',
    maxDepth: Number(cfg.maxDepth || 2),
    maxPackBytes: Number(cfg.maxPackBytes || 45000),
    maxGraphEdges: Number(cfg.maxGraphEdges || 200),
    maxSnippets: Number(cfg.maxSnippets || 40),
    fallbackToFullContext: cfg.fallbackToFullContext !== false,
  };
}

function extToLang(filePath) {
  const ext = path.extname(filePath).toLowerCase();
  if (ext === '.ts' || ext === '.tsx') return 'ts';
  if (ext === '.js' || ext === '.jsx' || ext === '.mjs' || ext === '.cjs') return 'js';
  if (ext === '.py') return 'py';
  if (ext === '.java') return 'java';
  if (ext === '.cs') return 'cs';
  return null;
}

function lineOfOffset(text, offset) {
  return text.slice(0, offset).split('\n').length;
}

function signatureLine(text, lineNum) {
  const lines = text.split('\n');
  return (lines[lineNum - 1] || '').trim().slice(0, 300);
}

function extractSymbols(filePath, content) {
  const lang = extToLang(filePath);
  if (!lang || !SYMBOL_PATTERNS[lang]) return [];

  const symbols = [];
  for (const pattern of SYMBOL_PATTERNS[lang]) {
    pattern.re.lastIndex = 0;
    let m;
    while ((m = pattern.re.exec(content)) !== null) {
      const name = m[1];
      const startLine = lineOfOffset(content, m.index);
      symbols.push({
        id: `${filePath}:${name}:${startLine}`,
        name,
        type: pattern.type,
        file: filePath,
        startLine,
        endLine: Math.min(startLine + 80, content.split('\n').length),
        signature: signatureLine(content, startLine),
      });
    }
  }
  return symbols;
}

function extractEdges(filePath, content, knownNames) {
  const edges = [];
  const localWords = new Set(content.match(/[A-Za-z_$][\w$]*/g) || []);
  for (const name of knownNames) {
    if (localWords.has(name)) {
      edges.push({ fromFile: filePath, toSymbol: name, kind: 'ref' });
    }
  }
  return edges;
}

function hashContent(s) {
  return crypto.createHash('md5').update(s).digest('hex');
}

function buildCodeIndex({ rootDir, files, previousIndex = null, maxFileSize = 30000 }) {
  const byFile = {};
  const allSymbols = [];

  for (const rel of files) {
    const abs = path.join(rootDir, rel);
    if (!fs.existsSync(abs)) continue;
    const st = fs.statSync(abs);
    if (st.size > maxFileSize) continue;
    const content = fs.readFileSync(abs, 'utf8');
    const fileHash = hashContent(content);

    const prev = previousIndex?.byFile?.[rel];
    if (prev && prev.hash === fileHash) {
      byFile[rel] = prev;
      allSymbols.push(...(prev.symbols || []));
      continue;
    }

    const symbols = extractSymbols(rel, content);
    byFile[rel] = { hash: fileHash, symbols, size: st.size };
    allSymbols.push(...symbols);
  }

  const knownNames = new Set(allSymbols.map(s => s.name));
  const allEdges = [];

  for (const rel of Object.keys(byFile)) {
    const abs = path.join(rootDir, rel);
    if (!fs.existsSync(abs)) continue;
    const content = fs.readFileSync(abs, 'utf8');
    allEdges.push(...extractEdges(rel, content, knownNames));
  }

  return {
    version: 1,
    generatedAt: new Date().toISOString(),
    symbols: allSymbols,
    edges: allEdges,
    byFile,
  };
}

function loadCodeIndex(indexPath) {
  if (!fs.existsSync(indexPath)) return null;
  try {
    return JSON.parse(fs.readFileSync(indexPath, 'utf8'));
  } catch {
    return null;
  }
}

function saveCodeIndex(indexPath, index) {
  fs.writeFileSync(indexPath, JSON.stringify(index, null, 2));
}

module.exports = {
  resolveIndexConfig,
  loadCodeIndex,
  saveCodeIndex,
  buildCodeIndex,
};

const STOP = new Set([
  'и', 'в', 'на', 'по', 'с', 'к', 'для', 'как', 'чтобы',
  'the', 'a', 'to', 'of', 'in', 'for', 'and', 'or', 'with', 'fix', 'add', 'bug'
]);

function tokens(text) {
  return (text || '')
    .toLowerCase()
    .match(/[a-zа-я0-9_$.:-]{2,}/gi)
    ?.filter(t => !STOP.has(t)) || [];
}

function scoreSymbols(promptTokens, symbols) {
  return symbols.map(sym => {
    let score = 0;
    for (const t of promptTokens) {
      if (sym.name.toLowerCase() === t) score += 10;
      else if (sym.name.toLowerCase().includes(t)) score += 4;
      else if ((sym.signature || '').toLowerCase().includes(t)) score += 2;
      else if ((sym.file || '').toLowerCase().includes(t)) score += 1;
    }
    return { sym, score };
  }).sort((a, b) => b.score - a.score);
}

function expand(index, seeds, maxDepth, maxEdges) {
  const selected = new Map();
  const queue = seeds.map(s => ({ name: s.name, depth: 0 }));
  const edges = [];

  while (queue.length) {
    const cur = queue.shift();
    if (selected.has(cur.name)) continue;

    for (const s of index.symbols) {
      if (s.name === cur.name) selected.set(s.id, s);
    }

    if (cur.depth >= maxDepth) continue;
    for (const e of index.edges) {
      if (e.toSymbol !== cur.name) continue;
      edges.push(e);
      if (edges.length >= maxEdges) break;
      queue.push({ name: e.toSymbol, depth: cur.depth + 1 });
    }
    if (edges.length >= maxEdges) break;
  }

  return { symbols: [...selected.values()], edges };
}

function readRange(rootDir, relPath, fromLine, toLine, redactSecrets) {
  const abs = path.join(rootDir, relPath);
  if (!fs.existsSync(abs)) return '';
  const lines = fs.readFileSync(abs, 'utf8').split('\n');
  const from = Math.max(1, fromLine);
  const to = Math.min(lines.length, toLine);
  const body = lines.slice(from - 1, to).join('\n');
  return redactSecrets ? redactSecrets(body) : body;
}

function dedupeByFileLine(symbols) {
  const map = new Map();
  for (const s of symbols) {
    const k = `${s.file}:${s.startLine}:${s.endLine}`;
    if (!map.has(k)) map.set(k, s);
  }
  return [...map.values()];
}

function buildContextPack({ rootDir, promptText, index, cfg, redactSecrets }) {
  if (!promptText || !index || !Array.isArray(index.symbols)) {
    return { used: false, reason: 'no prompt or no index' };
  }

  const promptTokens = tokens(promptText);
  const scored = scoreSymbols(promptTokens, index.symbols).filter(x => x.score > 0);
  const seeds = scored.slice(0, Math.max(4, Math.min(10, cfg.maxSnippets))).map(x => x.sym);

  if (!seeds.length) return { used: false, reason: 'no matching symbols' };

  const graph = expand(index, seeds, cfg.maxDepth, cfg.maxGraphEdges);
  const selectedSymbols = dedupeByFileLine(graph.symbols).slice(0, cfg.maxSnippets);

  const out = [];
  out.push('## CONTEXT PACK');
  out.push(`Prompt: ${promptText}`);
  out.push('');
  out.push('### Symbols');
  for (const s of selectedSymbols) {
    out.push(`- ${s.name} [${s.type}] ${s.file}:${s.startLine}-${s.endLine}`);
  }

  out.push('');
  out.push('### Graph (trimmed)');
  for (const e of graph.edges.slice(0, cfg.maxGraphEdges)) {
    out.push(`- ${e.fromFile} -> ${e.toSymbol} (${e.kind})`);
  }

  out.push('');
  out.push('### Relevant Code Fragments');

  let bytes = Buffer.byteLength(out.join('\n'), 'utf8');
  for (const s of selectedSymbols) {
    const from = Math.max(1, s.startLine - 6);
    const to = s.endLine + 6;
    const frag = readRange(rootDir, s.file, from, to, redactSecrets);

    const block = [
      `#### ${s.name} (${s.file}:${from}-${to})`,
      '```' + (path.extname(s.file).slice(1) || 'text'),
      frag,
      '```',
      ''
    ].join('\n');

    const nextBytes = bytes + Buffer.byteLength(block, 'utf8');
    if (nextBytes > cfg.maxPackBytes) break;
    out.push(block);
    bytes = nextBytes;
  }

  return {
    used: true,
    markdown: out.join('\n'),
    selectedCount: selectedSymbols.length,
  };
}

module.exports = {
  buildContextPack,
};

function readFile(filePath, options = {}) {
  const {
    fromLine = 1,
    toLine = Number.MAX_SAFE_INTEGER,
    maxBytes = Number.MAX_SAFE_INTEGER,
  } = options;

  try {
    const abs = path.join(process.cwd(), filePath);
    const raw = fs.readFileSync(abs, 'utf8');
    const lines = raw.split('\n');
    const sliced = lines.slice(Math.max(0, fromLine - 1), Math.min(lines.length, toLine)).join('\n');
    const limited = Buffer.byteLength(sliced, 'utf8') > maxBytes
      ? sliced.slice(0, maxBytes) + '\n... [truncated by maxBytes]'
      : sliced;
    return redactSecrets(limited);
  } catch (e) {
    return `[Error reading ${filePath}]`;
  }
}

5.4 Добавить sliding window для логов:

function readAiLogWindow(filePath, maxEntries = 10, maxBytes = 15000) {
  const abs = path.join(process.cwd(), filePath);
  if (!fs.existsSync(abs)) return `[Error reading ${filePath}]`;

  const text = fs.readFileSync(abs, 'utf8');
  const parts = text.split(/\n(?=## \[)/g);
  const lastEntries = parts.slice(-maxEntries).join('\n');
  const bytes = Buffer.byteLength(lastEntries, 'utf8');
  const trimmed = bytes > maxBytes
    ? lastEntries.slice(bytes - maxBytes)
    : lastEntries;

  return redactSecrets(trimmed);
}

function listIndexableFiles(treeCmd, maxFiles = 5000) {
  let raw = '';
  try {
    raw = execSync('git ls-files', { encoding: 'utf8' }).trim();
  } catch {
    raw = execSync(treeCmd, { encoding: 'utf8' }).trim();
  }

  return raw
    .split('\n')
    .map(s => s.trim())
    .filter(Boolean)
    .filter(f => /\.(ts|tsx|js|jsx|mjs|cjs|py|java|cs)$/i.test(f))
    .slice(0, maxFiles);
}

8) Риски реализации
==================================================

1. Regex даст ложные связи в сложном коде -> поэтому P1 с tree-sitter обязателен.
2. Слишком маленький `maxPackBytes` может обрезать нужный фрагмент -> нужен min floor (например 12KB).
3. Если graph слишком плотный, pack может заполниться шумом -> вводить cap на edges и snippets.

BUG-5 [MEDIUM]: Файлы читаются дважды (context-index.js, строки 195 и 216)

  buildCodeIndex читает каждый файл для extractSymbols, потом снова
  весь массив для extractEdges. Для 500 файлов — 1000 чтений вместо 500.

  Фикс: кэшировать содержимое в Map<relPath, string> при первом проходе.

BUG-6 [LOW]: redactSecrets — коллизия имён (context-pack.js, строка 314)

  Параметр redactSecrets в readRange — функция, переданная извне.
  Если передадут не функцию — `redactSecrets(body)` упадёт.

  Фикс: `return typeof redactSecrets === 'function' ? redactSecrets(body) : body;`

Пропущенные моменты:

1. Graceful degradation при битом индексе. loadCodeIndex возвращает null
   при ошибке, но buildContextPack обращается к index.symbols без
   null-check на вложенные поля. Невалидный .code_index.json — краш.

2. readPrompt вызывается дважды. В секции 5.6 до сборки бандла, и внутри

3. Инкрементальность частичная. Символы кэшируются по file hash — хорошо.
   Но рёбра пересчитываются с нуля каждый раз (строки 213-218).

4. listIndexableFiles использует git ls-files — новые untracked файлы

Итого: архитектура и декомпозиция готовы к реализации. Перед написанием
кода необходимо исправить BUG-1 (expand зацикливается) и BUG-2

влияют на качество, но не ломают работу.

10) Addendum by Gemini 3 Pro Preview (Stability & Quality)
==================================================

Вижу критические риски для production-стабильности, которые нужно закрыть в первой версии:

RISK-1 [CRITICAL]: ReDoS (Regex Denial of Service)
  Регулярные выражения в `SYMBOL_PATTERNS` могут вызвать "зависание" на специфических строках (минифицированный код, base64 data URI).
  Фикс: Обернуть `exec` в try-catch с таймаутом или добавить проверку длины строки перед парсингом (skip lines > 1000 chars).

RISK-2 [HIGH]: Atomic Writes (Corruption Prevention)
  Запись индекса происходит синхронно. Если процесс упадет посередине, файл будет битым.
  Фикс: Писать во временный файл `.code_index.json.tmp`, затем делать `fs.renameSync` (атомарная операция).

RISK-3 [MEDIUM]: Test Pollution
  Тестовые файлы (`.test.ts`, `.spec.js`) создают много "шумных" связей, которые редко нужны (кроме задачи "почини тест").
  Фикс: Добавить `ignorePatterns: ["**/*.test.*", "**/*.spec.*"]` в конфиг индексатора по умолчанию.

RISK-4 [MEDIUM]: Signature Trimming
  `signatureLine` берет просто 300 символов. Для сжатого JS/TS это захватит тело функции.
  Фикс: Обрезать сигнатуру по первому вхождению `{` или `;`.

RISK-5 [LOW]: Dangling References
  Если файл удален, но ссылка на него осталась в кэше `previousIndex`, граф будет битым.
  Фикс: При загрузке `previousIndex` проверять существование файлов. Если файл пропал — инвалидировать соответствующую часть графа.

Рекомендация: Включить эти фиксы (особенно Atomic Writes и ReDoS protection) сразу в Этап 1 (`ai/scripts/context-index.js`).

Вопрос 1: Согласен ли я с P0 gate из 5 пунктов без расширения scope?

Да, согласен. 5 пунктов покрывают все блокеры:
  1. BUG-1 (expand) — без него граф не работает
  2. BUG-2 (fallback) — без него context pack бесполезен
  3. BUG-4 (шум рёбер) — без него граф заполнен мусором
  4. Atomic write — без него крэш = битый индекс
  5. Untracked файлы — без него новые файлы невидимы

Оговорка по пункту 3 (BUG-4): фильтр "по import/use-site и стоп-имена"
— правильное направление, но для P0 достаточно минимального:
  - исключить символы короче 4 символов
  - исключить self-references (файл ссылается на символ, определённый в нём же)
Import-based фильтрация требует парсинга import-statements — это P1.

Scope расширять не нужно.

Вопрос 3: Готова ли спецификация к реализации P0 после этих правок?

Да, подтверждаю. После 5 обязательных фиксов спецификация готова.
